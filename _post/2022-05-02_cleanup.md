``` bash
find . -type f -empty  -not -path '*/\.git/*'; 
find . -type d -empty ! -name .git  -delete

find . -iname node_modules -exec rm -rfv "{}" \;
find . -iname .ipynb_checkpoints -exec  rm -rfv "{}" \; 
find . -iname __pycache__ -exec  rm -rfv "{}" \;
```
